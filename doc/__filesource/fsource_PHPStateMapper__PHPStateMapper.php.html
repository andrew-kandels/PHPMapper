  145.     /**

  245.     /**
  246.      * Similar to imagepng(). If no file is given, it prints headers and
  247.      * draws the image directly to the browser. If a file is given, it instead
  248.      * fills it with the contents of the image.
  249.      *
  250.      * @param   string      File name (or null, for output)
  251.      * @param   integer     Compression level (0-9)
  252.      * @return  boolean     Success?
  253.      * @throws  PHPStateMapper_Exception
  254.      */
  255.     public function draw($file = null, $compression = 0)
  256.     {
  257.         if (!$img = imagecreatefrompng($tmp = dirname(__FILE__) . '/images/united_states.png'))
  258.         {
  259.             throw new PHPStateMapper_Exception_Image("Failed to load $tmp");
  260.         }
  261.  
  262.         $width  = imagesx($img);
  263.         $height = imagesy($img);
  264.         $max    = $this->_getMaxValue();
  265.  
  266.         foreach ($this->_states as $name => $value)
  267.         {
  268.             if ($value <= 0) continue;
  269.  
  270.             if (!$subImg = imagecreatefrompng($name))
  271.             {
  272.                 throw new PHPStateMapper_Exception_Image("Failed to open $name");
  273.             }
  274.  
  275.             $index = imagecolorclosest($subImg, 236, 201, 79);
  276.             list($r, $g, $b) = $this->_color;
  277.             imagecolorset($subImg, $index, $r, $g, $b);
  278.  
  279.             // Make top/left pixel the transparent color (white)
  280.             imagecolortransparent($subImg, imagecolorat($subImg, 0, 0));
  281.  
  282.             // Color in the state most popular being darker, using an alpha
  283.             $pct = ceil($value / $max * 100);
  284.  
  285.             // Copy smaller image over the map
  286.             imagecopymerge($img, $subImg, 0, 0, 0, 0, $width, $height, $pct);
  287.         }
  288.  
  289.         if (!$file)
  290.         {
  291.             header('Content-type: image/png');
  292.         }
  293.  
  294.         if (!imagepng($img, $file, $compression))
  295.         {
  296.             throw new PHPStateMapper_Exception_Image("Failed to create $file");
  297.         }
  298.         else
  299.         {
  300.             return true;
  301.         }
  302.     }
